SELECT 
    us.User_ID,
    us.username,
    us.email,
    us.Role_ID,
    js.Jobseeker_ID,
    js.avatar AS jobseeker_avatar,
    js.Status_ AS jobseeker_status,
    pjs.Profile_ID,
    pjs.last_name,
    pjs.first_name,
    pjs.gender,
    pjs.Birthday,
    pjs.avatar AS profile_avatar,
    pjs.Career_target,
    pjs.Title AS profile_title,
    pjs.Marital_status,
    pjs.Nationality,
    pjs.Level_ID,
    pjs.Job_function_ID,
    pjs.Nation_ID,
    pjs.City_ID,
    pjs.District_ID,
    pjs.Street,
    cd.District_name,
    cc.City_name,
    GROUP_CONCAT(DISTINCT pe.Experience SEPARATOR ', ') AS experiences,
    GROUP_CONCAT(DISTINCT ped.Situation SEPARATOR ', ') AS situations
    FROM 
        User_ us
    JOIN Jobseeker js ON us.User_ID = js.Jobseeker_ID
    JOIN Profile_jobseeker pjs ON js.Jobseeker_ID = pjs.Profile_ID
    LEFT JOIN Catalog_District cd ON pjs.District_ID = cd.District_ID
    LEFT JOIN Catalog_City cc ON pjs.City_ID = cc.City_ID
    LEFT JOIN Profile_Experience pe ON pjs.Profile_ID = pe.Profile_ID
    LEFT JOIN Profile_Education ped ON pjs.Profile_ID = ped.Profile_ID
    WHERE 
        us.User_ID = ?  
    GROUP BY 
        us.User_ID, js.Jobseeker_ID, pjs.Profile_ID, cd.District_name, cc.City_name;
